% --- Figure placeholder macro ------------------------------------------
% Renders the file if it exists (PDF preferred, then PNG, then JPG),
% otherwise a visible framed box so the LaTeX compiles before real
% images arrive.
% Usage: \figplaceholder{filename}{width}{caption}{label}
\newcommand{\figplaceholder}[4]{%
  \begin{figure}[H]
    \centering
    \IfFileExists{figures/#1.pdf}%
      {\includegraphics[width=#2]{figures/#1.pdf}}%
      {\IfFileExists{figures/#1.png}%
        {\includegraphics[width=#2]{figures/#1.png}}%
        {\IfFileExists{figures/#1.jpg}%
          {\includegraphics[width=#2]{figures/#1.jpg}}%
          {\fbox{\parbox{#2}{\vspace{1.0cm}\centering%
            {\itshape\bfseries PLACEHOLDER: \path{figures/#1}}\par
            \medskip
            {\itshape (Figure to be inserted --- see caption below.)}\par
            \vspace{1.0cm}}}}}}%
    \caption{#3}
    \label{#4}
  \end{figure}}

% Non-floating variant: fixes the figure at its exact source position
% (via [H] from the float package). Used for the pipeline diagram so
% it anchors right under the Pipeline Overview subsection rather than
% drifting into a later subsection. Uses `figure` rather than
% `figure*` because [H] does not combine with `figure*`; in a
% single-column document the two are equivalent anyway.
\newcommand{\figplaceholderwide}[4]{%
  \begin{figure}[H]
    \centering
    \IfFileExists{figures/#1.pdf}%
      {\includegraphics[width=#2]{figures/#1.pdf}}%

          \medskip
          {\itshape (Figure to be inserted --- see caption below.)}\par
          \vspace{1.0cm}}}}}%
    \caption{#3}
    \label{#4}
  \end{figure}}
